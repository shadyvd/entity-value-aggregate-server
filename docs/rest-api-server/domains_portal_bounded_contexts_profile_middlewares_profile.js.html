import merge from 'lodash.merge';

import { EVASBaseFactory } from '@twyr/framework-classes';
import { createErrorForPropagation } from '@twyr/error-serializer';
import { BaseMiddleware } from 'baseclass:middleware';

/**
 * @class Profile
 * @extends BaseMiddleware
 *
 * @param {string} [location] - __dirname for this file in CJS, basically
 * @param {object} [domainInterface] - Domain functionality exposed to sub-artifacts
 *
 * @classdesc The Middleware to handle login / logout / register
 */
export class Profile extends BaseMiddleware {
	// #region Constructor
	// eslint-disable-next-line jsdoc/require-jsdoc
	constructor(location, domainInterface) {
		super(location, domainInterface);
	}
	// #endregion

	// #region Protected Methods, to be overridden by derived classes
	/**
	 * @memberof Profile
	 * @async
	 * @instance
	 * @override
	 * @function
	 * @name _registerApi
	 *
	 * @returns {null} - Nothing
	 *
	 * @description
	 * Adds the API to the apiRegistry in the domainInterface
	 *
	 */
	async _registerApi() {
		const errors = [];

		try {
			let registerResolutions = [];

			const baseApis = await super._registerApi?.();
			for (const baseApi of baseApis ?? []) {
				registerResolutions?.push?.(
					this?.domainInterface?.apiRegistry?.register?.(
						baseApi?.pattern,
						baseApi?.handler
					)
				);
			}

			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'CREATE',
					this.#createProfile?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'GET',
					this.#getProfile?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'SET',
					this.#setProfile?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'DELETE',
					this.#deleteProfile?.bind?.(this)
				)
			);

			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'CREATE_CONTACT',
					this.#createProfileContact?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'GET_ALL_CONTACTS',
					this.#getAllProfileContacts?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'GET_CONTACT',
					this.#getProfileContact?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'SET_CONTACT',
					this.#setProfileContact?.bind?.(this)
				)
			);
			registerResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.register?.(
					'DELETE_CONTACT',
					this.#deleteProfileContact?.bind?.(this)
				)
			);

			registerResolutions =
				await Promise?.allSettled?.(registerResolutions);
			for (const registerResolution of registerResolutions) {
				if (registerResolution?.status === 'fulfilled') continue;
				errors?.push?.(registerResolution?.reason);
			}
		} catch (error) {
			errors?.push?.(error);
		}

		if (!errors?.length) return;

		const propagatedError = createErrorForPropagation?.(
			`${this?.name}::_registerApi error`,
			errors
		);

		if (propagatedError) throw propagatedError;
	}

	/**
	 * @memberof Profile
	 * @async
	 * @instance
	 * @override
	 * @function
	 * @name _unregisterApi
	 *
	 * @returns {null} - Nothing
	 *
	 * @description
	 * Removes the API from the apiRegistry in the domainInterface
	 *
	 */
	async _unregisterApi() {
		const errors = [];

		try {
			let unregisterResolutions = [];

			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'DELETE_CONTACT',
					this.#deleteProfileContact?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'SET_CONTACT',
					this.#setProfileContact?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'GET_CONTACT',
					this.#getProfileContact?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'GET_ALL_CONTACTS',
					this.#getAllProfileContacts?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'CREATE_CONTACT',
					this.#createProfileContact?.bind?.(this)
				)
			);

			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'DELETE',
					this.#deleteProfile?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'SET',
					this.#setProfile?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'GET',
					this.#getProfile?.bind?.(this)
				)
			);
			unregisterResolutions?.push?.(
				this?.domainInterface?.apiRegistry?.unregister?.(
					'CREATE',
					this.#createProfile?.bind?.(this)
				)
			);

			unregisterResolutions = await Promise?.allSettled?.(
				unregisterResolutions
			);
			for (const unregisterResolution of unregisterResolutions) {
				if (unregisterResolution?.status === 'fulfilled') continue;
				errors?.push?.(unregisterResolution?.reason);
			}

			await super._unregisterApi?.();
		} catch (error) {
			errors?.push?.(error);
		}

		if (!errors?.length) return;

		const propagatedError = createErrorForPropagation?.(
			`${this?.name}::_unregisterApi error`,
			errors
		);

		if (propagatedError) throw propagatedError;
	}
	// #endregion

	// #region Handlers
	async #createProfile({ data }) {
		// Step 0: Get the models
		const UserModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user'
			}
		]);

		// Step 1: De-serialize from JSON API Format, and generate
		// unique password
		const userProfile =
			await this?.domainInterface?.serializer?.deserializeAsync?.(
				data?.data?.type,
				data
			);

		let generatePassword = await import('generate-password');
		generatePassword = generatePassword?.['default'];

		const generatedPassword = generatePassword?.generate?.({
			numbers: true,
			symbols: true,
			lowercase: true,
			uppercase: true,
			strict: true
		});

		let upash = await import('upash');
		upash = upash?.['default'];

		userProfile.password = await upash?.hash?.(generatedPassword);

		delete userProfile?.id;
		delete userProfile?.created_at;
		delete userProfile?.updated_at;

		// Step 2: Insert into the database
		let createdProfile = await this?._executeWithBackOff?.(() => {
			return UserModel?.query?.()?.insertAndFetch?.(userProfile);
		});

		delete createdProfile?.password;

		// Step 3: Send notification with username / password
		const emailBodyData = merge?.({}, createdProfile, {
			password: generatedPassword
		});

		const newUserTemplate = await this?._getTemplatesFromDomain?.([
			{
				action: 'new-user',
				type: 'email'
			}
		]);

		const renderer =
			await this?.domainInterface?.iocContainer?.resolve?.('Renderer');
		let emailBody = await renderer?.render?.([
			{
				template: newUserTemplate,
				data: emailBodyData
			}
		]);

		const notification =
			await this?.domainInterface?.iocContainer?.resolve?.(
				'Notification'
			);

		emailBody = emailBody?.[0];
		await notification?.send?.([
			{
				type: 'email',
				to: createdProfile?.email,
				subject: 'XXX',
				body: emailBody?.document
			}
		]);

		// Step 4: Serialize to JSON API Format
		createdProfile =
			await this?.domainInterface?.serializer?.serializeAsync?.(
				'user',
				createdProfile
			);

		// Finally, return..
		return {
			status: 200,
			body: createdProfile
		};
	}

	async #getProfile({ user }) {
		let UserModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user'
			}
		]);

		let userProfile = await this?._executeWithBackOff?.(() => {
			return UserModel?.query?.()
				?.findById?.(user?.id)
				?.withGraphFetched?.('contacts.[contactType]');
		});

		// Serialize to JSON API Format
		userProfile = await this?.domainInterface?.serializer?.serializeAsync?.(
			'user',
			userProfile
		);

		return {
			status: 200,
			body: userProfile
		};
	}

	async #setProfile({ data }) {
		let UserModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user'
			}
		]);

		// Step 1: De-serialize from JSON API Format, and remove non-update-able
		// fields
		const userProfile =
			await this?.domainInterface?.serializer?.deserializeAsync?.(
				data?.data?.type,
				data
			);

		// TODO: Security Issue here... do a proper check
		// if (!user?.['is_superuser'] && user?.id !== userProfile?.id) {
		// 	throw new Error(
		// 		`${user?.['first_name']} ${user?.['last_name']} cannot update details of ${userProfile?.['id']}`
		// 	);
		// }

		const userProfileId = userProfile?.id;

		delete userProfile?.id;
		delete userProfile?.email;
		delete userProfile?.password;
		delete userProfile?.created_at;
		delete userProfile?.updated_at;

		// Step 2: Update the database
		let updatedPerson = await this?._executeWithBackOff?.(() => {
			return UserModel?.query?.()?.patchAndFetchById?.(
				userProfileId,
				userProfile
			);
		});

		// Step 3: Send the notification
		const updatedUserTemplate = await this?._getTemplatesFromDomain?.([
			{
				action: 'update-user',
				type: 'email'
			}
		]);

		const renderer =
			await this?.domainInterface?.iocContainer?.resolve?.('Renderer');
		let emailBody = await renderer?.render?.([
			{
				template: updatedUserTemplate,
				data: updatedPerson
			}
		]);

		emailBody = emailBody?.[0];
		const notification =
			await this?.domainInterface?.iocContainer?.resolve?.(
				'Notification'
			);
		await notification?.send?.([
			{
				type: 'email',
				to: updatedPerson?.email,
				subject: 'XXX',
				body: emailBody?.document
			}
		]);

		// Finally... return
		updatedPerson =
			await this?.domainInterface?.serializer?.serializeAsync?.(
				'user',
				updatedPerson
			);

		return {
			status: 200,
			body: updatedPerson
		};
	}

	async #deleteProfile({ user }) {
		let UserModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user'
			}
		]);

		await this?._executeWithBackOff?.(() => {
			return UserModel?.query?.()?.deleteById?.(user?.id);
		});

		return {
			status: 204
		};
	}

	async #createProfileContact({ user, data }) {
		// Step 0: Get the user contact model
		let UserContactModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user-contact'
			}
		]);

		// Step 1: De-serialize from JSON API Format
		const userContact =
			await this?.domainInterface?.serializer?.deserializeAsync?.(
				data?.data?.type,
				data
			);

		delete userContact?.id;
		delete userContact?.created_at;
		delete userContact?.updated_at;

		if (userContact?.['user_id'] !== user?.id)
			throw new Error('User cannot create contacts for another user');

		// Step 2: Insert into the database
		let createdContact = await this?._executeWithBackOff?.(() => {
			return UserContactModel?.query?.()?.insertAndFetch?.(userContact);
		});

		// Step 3: Serialize to JSON API Format
		createdContact =
			await this?.domainInterface?.serializer?.serializeAsync?.(
				'user_contact',
				createdContact
			);

		// Finally, return..
		return {
			status: 200,
			body: createdContact
		};
	}
	async #getAllProfileContacts({ user }) {
		// Step 1: Fetch the user contact model...
		let UserContactModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user-contact'
			}
		]);

		// Step 2: Fetch the contacts
		let userContacts = await this?._executeWithBackOff?.(() => {
			return UserContactModel?.query?.()
				?.where?.('user_id', '=', user?.id)
				?.withGraphFetched?.('contactType');
		});

		// Step 3: Serialize to JSON API Format
		userContacts =
			await this?.domainInterface?.serializer?.serializeAsync?.(
				'user_contact',
				userContacts
			);

		// Finally, return...
		return {
			status: 200,
			body: userContacts
		};
	}
	async #getProfileContact({ user, contactId }) {
		// Step 1: Get the user contact model
		let UserContactModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user-contact'
			}
		]);

		// Step 2: Get data from the database
		let userContact = await this?._executeWithBackOff?.(() => {
			return UserContactModel?.query?.()
				?.findById?.(contactId)
				?.withGraphFetched?.('contactType');
		});

		if (userContact?.['user_id'] !== user?.id)
			throw new Error(`User cannot query contacts of another user`);

		// Finally, serialize to JSON API Format and return
		userContact = await this?.domainInterface?.serializer?.serializeAsync?.(
			'user_contact',
			userContact
		);

		return {
			status: 200,
			body: userContact
		};
	}
	async #setProfileContact({ data, user, contactId }) {
		// Step 1: Sanity checks
		if (user?.['id'] !== data?.data?.attributes?.['user_id'])
			throw new Error(`User cannot modify a contact of another user`);

		if (contactId !== data?.data?.['id'])
			throw new Error(`Data does not match the contact id`);

		// Step 2: Get the tenant model
		let UserContactModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user-contact'
			}
		]);

		// Step 3: De-serialize from JSON API Format, and remove non-update-able
		// fields
		const userContact =
			await this?.domainInterface?.serializer?.deserializeAsync?.(
				data?.data?.type,
				data
			);
		delete userContact?.id;
		delete userContact?.created_at;
		delete userContact?.updated_at;

		// Step 4: Update the database
		let updatedContact = await this?._executeWithBackOff?.(() => {
			return UserContactModel?.query?.()?.patchAndFetchById?.(
				contactId,
				userContact
			);
		});

		updatedContact =
			await this?.domainInterface?.serializer?.serializeAsync?.(
				'user_contact',
				updatedContact
			);

		// Finally... return
		return {
			status: 200,
			body: updatedContact
		};
	}
	async #deleteProfileContact({ user, contactId }) {
		// Step 1: Get the model
		let UserContactModel = await this?._getModelsFromDomain?.([
			{
				type: 'relational',
				name: 'user-contact'
			}
		]);

		// Step 2: Delete
		await this?._executeWithBackOff?.(() => {
			return UserContactModel?.query?.()
				?.delete?.()
				?.whereRaw?.(`id = ? AND user_id = ?`, [contactId, user?.id]);
		});

		return {
			status: 204
		};
	}
	// #endregion
}

/**
 * @class MiddlewareFactory
 * @extends EVASBaseFactory
 *
 * @classdesc The Profile Module Profile Middleware Class Factory.
 */
export default class MiddlewareFactory extends EVASBaseFactory {
	// #region Constructor
	// eslint-disable-next-line jsdoc/require-jsdoc
	constructor() {
		super();
	}
	// #endregion

	// #region Lifecycle API
	/**
	 * @memberof MiddlewareFactory
	 * @async
	 * @static
	 * @override
	 * @function
	 * @name createInstance
	 *
	 * @param {object} [domainInterface] - Domain functionality exposed to sub-artifacts
	 *
	 * @returns {Profile} - The Profile middleware instance.
	 *
	 */
	static async createInstances(domainInterface) {
		if (!MiddlewareFactory.#profileInstance) {
			const profileInstance = new Profile(
				MiddlewareFactory['$disk_unc'],
				domainInterface
			);

			await profileInstance?.load?.();
			MiddlewareFactory.#profileInstance = profileInstance;
		}

		return MiddlewareFactory.#profileInstance;
	}

	/**
	 * @memberof MiddlewareFactory
	 * @async
	 * @static
	 * @override
	 * @function
	 * @name destroyInstances
	 *
	 * @returns {undefined} - Nothing.
	 *
	 * @description Clears the Profile instance
	 */
	static async destroyInstances() {
		await MiddlewareFactory.#profileInstance?.unload?.();
		MiddlewareFactory.#profileInstance = undefined;

		return;
	}
	// #endregion

	// #region Getters
	/**
	 * @memberof MiddlewareFactory
	 * @async
	 * @static
	 * @override
	 * @function
	 * @name MiddlewareName
	 *
	 * @returns {string} - Name of this middleware.
	 *
	 * @description
	 * Returns the name of this middleware - Profile
	 */
	static get MiddlewareName() {
		return 'Profile';
	}
	// #endregion

	// #region Private Static Members
	static #profileInstance = undefined;
	// #endregion
}
